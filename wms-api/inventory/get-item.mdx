---
title: "Get Inventory Item"
api: "GET /wms/inventory/{item_id}"
description: "Get inventory item by ID"
---

### Headers

<ParamField header="x-trackstar-api-key" type="string" required>
  Your organization-level Trackstar API Key.
</ParamField>

<ParamField header="x-trackstar-access-token" type="string" required>
  Your user's access token for a specific integration (ShipHero, Extensiv, etc).
</ParamField>

### Path

<ParamField path="item_id" type="string" required>
  The inventory item ID you want to query.
</ParamField>

### Response

<ResponseField name="data" type="object">

An inventory item.

<Expandable title="Inventory Item" defaultOpen="true">

<ResponseField name="id" type="string">
  The unique ID of an inventory item.
</ResponseField>

<ResponseField name="name" type="string">
</ResponseField>

<ResponseField name="active" type="boolean">
</ResponseField>

<ResponseField name="onhand" type="integer">
  The total amount of inventory in the warehouse.
</ResponseField>

<ResponseField name="committed" type="integer">
  Inventory that is already assigned to orders.
</ResponseField>

<ResponseField name="fulfillable" type="integer">
  Inventory that can be fulfilled for orders.
</ResponseField>

<ResponseField name="exception" type="integer">
  Inventory that is in exception status.
</ResponseField>

<ResponseField name="sellable" type="integer">
  Inventory that can be sold.
</ResponseField>

<ResponseField name="awaiting" type="integer">
  Inventory that the warehouse is awaiting delivery of.
</ResponseField>

</Expandable>

</ResponseField>
