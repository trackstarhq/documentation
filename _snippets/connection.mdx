<Expandable title="Connection">

<ResponseField name="connection_id" type="string">
  The unique ID of the connection. (UUID)
</ResponseField>

<ResponseField name="integration_name" type="string">
  The name of the integration associated with the connection.
</ResponseField>

<ResponseField name="created_at" type="string">
  The date the connection was created. Returned in YYYY-MM-DDTHH:MM:SSZ format.
</ResponseField>

<ResponseField name="last_used" type="string or null">
  The date the connection was last used in a request to the Trackstar API. Returned in YYYY-MM-DDTHH:MM:SSZ format.
</ResponseField>

<ResponseField name="times_used" type="integer or null">
  The number of times the connection has been used in a request to the Trackstar API. (Aka # of API calls made with this connection)
</ResponseField>

<ResponseField name="errors" type="list of errors">

    <ResponseField name="error_message" type="string">

    </ResponseField>
    <ResponseField name="first_seen" type="string">

    </ResponseField>
    <ResponseField name="last_seen" type="string">
      The date the error was last seen. Returned in YYYY-MM-DDTHH:MM:SSZ format.
    </ResponseField>
  </Expandable>
